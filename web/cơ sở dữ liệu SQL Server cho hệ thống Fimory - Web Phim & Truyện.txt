-- =============================================
-- FIMORY - HỆ THỐNG WEB PHIM & TRUYỆN
-- Database: SQL Server
-- Version: 1.0
-- =============================================

USE master;
GO

-- Tạo database
IF NOT EXISTS (SELECT * FROM sys.databases WHERE name = 'Fimory')
BEGIN
    CREATE DATABASE Fimory;
END
GO

USE Fimory;
GO

-- =============================================
-- 1. BẢNG NGƯỜI DÙNG & PHÂN QUYỀN
-- =============================================

-- Bảng vai trò
CREATE TABLE Roles (
    RoleID INT PRIMARY KEY IDENTITY(1,1),
    RoleName NVARCHAR(50) NOT NULL UNIQUE, -- Viewer, Uploader, Author, Translator, Reup, Admin
    Description NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- Bảng người dùng
CREATE TABLE Users (
    UserID INT PRIMARY KEY IDENTITY(1,1),
    Username NVARCHAR(50) NOT NULL UNIQUE,
    Email NVARCHAR(100) NOT NULL UNIQUE,
    PasswordHash NVARCHAR(255) NOT NULL,
    FullName NVARCHAR(100),
    Avatar NVARCHAR(500),
    RoleID INT NOT NULL DEFAULT 1,
    IsEmailVerified BIT DEFAULT 0,
    IsActive BIT DEFAULT 1,
    IsVIP BIT DEFAULT 0,
    VIPExpiryDate DATETIME NULL,
    Points INT DEFAULT 0, -- Điểm thưởng
    FailedLoginAttempts INT DEFAULT 0,
    LockedUntil DATETIME NULL,
    LastLoginAt DATETIME,
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (RoleID) REFERENCES Roles(RoleID)
);

-- Bảng OTP xác thực email
CREATE TABLE EmailOTP (
    OTPID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    OTPCode NVARCHAR(6) NOT NULL,
    ExpiryTime DATETIME NOT NULL,
    IsUsed BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE
);

-- Bảng yêu cầu nâng cấp quyền
CREATE TABLE RoleUpgradeRequests (
    RequestID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    RequestedRoleID INT NOT NULL,
    Reason NVARCHAR(500),
    Status NVARCHAR(20) DEFAULT 'Pending', -- Pending, Approved, Rejected
    ReviewedBy INT NULL,
    ReviewNote NVARCHAR(500),
    RequestedAt DATETIME DEFAULT GETDATE(),
    ReviewedAt DATETIME NULL,
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (RequestedRoleID) REFERENCES Roles(RoleID),
    FOREIGN KEY (ReviewedBy) REFERENCES Users(UserID)
);

-- =============================================
-- 2. BẢNG DANH MỤC & THỂ LOẠI
-- =============================================

-- Bảng thể loại (có thể dùng chung cho phim & truyện)
CREATE TABLE Categories (
    CategoryID INT PRIMARY KEY IDENTITY(1,1),
    CategoryName NVARCHAR(100) NOT NULL,
    Slug NVARCHAR(100) NOT NULL UNIQUE,
    Description NVARCHAR(500),
    Type NVARCHAR(20) NOT NULL, -- Movie, Series, Both
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- Bảng tag
CREATE TABLE Tags (
    TagID INT PRIMARY KEY IDENTITY(1,1),
    TagName NVARCHAR(50) NOT NULL UNIQUE,
    Slug NVARCHAR(50) NOT NULL UNIQUE,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- =============================================
-- 3. BẢNG PHIM
-- =============================================

-- Bảng phim
CREATE TABLE Movies (
    MovieID INT PRIMARY KEY IDENTITY(1,1),
    Title NVARCHAR(255) NOT NULL,
    Slug NVARCHAR(255) NOT NULL UNIQUE,
    Description NVARCHAR(MAX),
    PosterURL NVARCHAR(500),
    TrailerURL NVARCHAR(500),
    ReleaseYear INT,
    Season INT NULL, -- Mùa phim (nếu có)
    Duration INT, -- Thời lượng (phút)
    Country NVARCHAR(100),
    Director NVARCHAR(255),
    Cast NVARCHAR(MAX), -- Diễn viên (lưu dạng JSON hoặc text)
    Status NVARCHAR(20) DEFAULT 'Ongoing', -- Ongoing, Completed
    IsFree BIT DEFAULT 1, -- Miễn phí hay trả phí
    ViewCount BIGINT DEFAULT 0,
    Rating DECIMAL(3,2) DEFAULT 0.00, -- Đánh giá trung bình
    TotalRatings INT DEFAULT 0,
    Language NVARCHAR(50) DEFAULT 'vi', -- Đa ngôn ngữ
    UploaderID INT NOT NULL,
    IsApproved BIT DEFAULT 0, -- Admin duyệt
    ApprovedBy INT NULL,
    ApprovedAt DATETIME NULL,
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UploaderID) REFERENCES Users(UserID),
    FOREIGN KEY (ApprovedBy) REFERENCES Users(UserID)
);

-- Bảng tập phim
CREATE TABLE MovieEpisodes (
    EpisodeID INT PRIMARY KEY IDENTITY(1,1),
    MovieID INT NOT NULL,
    EpisodeNumber INT NOT NULL,
    Title NVARCHAR(255),
    VideoURL NVARCHAR(500) NOT NULL,
    Duration INT, -- Thời lượng tập phim
    IsFree BIT DEFAULT 1,
    ViewCount BIGINT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,
    UNIQUE(MovieID, EpisodeNumber)
);

-- Bảng liên kết phim - thể loại
CREATE TABLE MovieCategories (
    MovieID INT NOT NULL,
    CategoryID INT NOT NULL,
    PRIMARY KEY (MovieID, CategoryID),
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,
    FOREIGN KEY (CategoryID) REFERENCES Categories(CategoryID) ON DELETE CASCADE
);

-- Bảng liên kết phim - tag
CREATE TABLE MovieTags (
    MovieID INT NOT NULL,
    TagID INT NOT NULL,
    PRIMARY KEY (MovieID, TagID),
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,
    FOREIGN KEY (TagID) REFERENCES Tags(TagID) ON DELETE CASCADE
);

-- =============================================
-- 4. BẢNG TRUYỆN
-- =============================================

-- Bảng truyện
CREATE TABLE Series (
    SeriesID INT PRIMARY KEY IDENTITY(1,1),
    Title NVARCHAR(255) NOT NULL,
    Slug NVARCHAR(255) NOT NULL UNIQUE,
    Description NVARCHAR(MAX),
    CoverURL NVARCHAR(500),
    Author NVARCHAR(255),
    Status NVARCHAR(20) DEFAULT 'Ongoing', -- Ongoing, Completed
    IsFree BIT DEFAULT 1,
    ViewCount BIGINT DEFAULT 0,
    Rating DECIMAL(3,2) DEFAULT 0.00,
    TotalRatings INT DEFAULT 0,
    Language NVARCHAR(50) DEFAULT 'vi',
    UploaderID INT NOT NULL,
    IsApproved BIT DEFAULT 0,
    ApprovedBy INT NULL,
    ApprovedAt DATETIME NULL,
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UploaderID) REFERENCES Users(UserID),
    FOREIGN KEY (ApprovedBy) REFERENCES Users(UserID)
);

-- Bảng chapter truyện
CREATE TABLE Chapters (
    ChapterID INT PRIMARY KEY IDENTITY(1,1),
    SeriesID INT NOT NULL,
    ChapterNumber INT NOT NULL,
    Title NVARCHAR(255),
    Content NVARCHAR(MAX), -- Nội dung chapter
    IsFree BIT DEFAULT 1,
    ViewCount BIGINT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    UNIQUE(SeriesID, ChapterNumber)
);

-- Bảng liên kết truyện - thể loại
CREATE TABLE SeriesCategories (
    SeriesID INT NOT NULL,
    CategoryID INT NOT NULL,
    PRIMARY KEY (SeriesID, CategoryID),
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    FOREIGN KEY (CategoryID) REFERENCES Categories(CategoryID) ON DELETE CASCADE
);

-- Bảng liên kết truyện - tag
CREATE TABLE SeriesTags (
    SeriesID INT NOT NULL,
    TagID INT NOT NULL,
    PRIMARY KEY (SeriesID, TagID),
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    FOREIGN KEY (TagID) REFERENCES Tags(TagID) ON DELETE CASCADE
);

-- =============================================
-- 5. BẢNG TƯƠNG TÁC NGƯỜI DÙNG
-- =============================================

-- Bảng yêu thích phim
CREATE TABLE MovieFavorites (
    FavoriteID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    MovieID INT NOT NULL,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,
    UNIQUE(UserID, MovieID)
);

-- Bảng yêu thích truyện
CREATE TABLE SeriesFavorites (
    FavoriteID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    SeriesID INT NOT NULL,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    UNIQUE(UserID, SeriesID)
);

-- Bảng đánh giá phim
CREATE TABLE MovieRatings (
    RatingID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    MovieID INT NOT NULL,
    Rating INT CHECK (Rating >= 1 AND Rating <= 5),
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,
    UNIQUE(UserID, MovieID)
);

-- Bảng đánh giá truyện
CREATE TABLE SeriesRatings (
    RatingID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    SeriesID INT NOT NULL,
    Rating INT CHECK (Rating >= 1 AND Rating <= 5),
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    UNIQUE(UserID, SeriesID)
);

-- Bảng bình luận phim
CREATE TABLE MovieComments (
    CommentID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    MovieID INT NOT NULL,
    ParentCommentID INT NULL, -- Trả lời bình luận
    Content NVARCHAR(1000) NOT NULL,
    IsDeleted BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID),
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,
    FOREIGN KEY (ParentCommentID) REFERENCES MovieComments(CommentID)
);

-- Bảng bình luận truyện
CREATE TABLE SeriesComments (
    CommentID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    SeriesID INT NOT NULL,
    ParentCommentID INT NULL,
    Content NVARCHAR(1000) NOT NULL,
    IsDeleted BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID),
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    FOREIGN KEY (ParentCommentID) REFERENCES SeriesComments(CommentID)
);

-- Bảng lịch sử xem phim
CREATE TABLE MovieHistory (
    HistoryID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    MovieID INT NOT NULL,
    EpisodeID INT NULL,
    WatchedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE NO ACTION,
    FOREIGN KEY (EpisodeID) REFERENCES MovieEpisodes(EpisodeID) ON DELETE NO ACTION
);

-- Bảng lịch sử đọc truyện
CREATE TABLE SeriesHistory (
    HistoryID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    SeriesID INT NOT NULL,
    ChapterID INT NULL,
    ReadAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE NO ACTION,
    FOREIGN KEY (ChapterID) REFERENCES Chapters(ChapterID) ON DELETE NO ACTION
);

-- =============================================
-- 6. BẢNG BÁO CÁO VI PHẠM
-- =============================================

CREATE TABLE Reports (
    ReportID INT PRIMARY KEY IDENTITY(1,1),
    ReporterID INT NOT NULL,
    ContentType NVARCHAR(20) NOT NULL, -- Movie, Series, Comment
    ContentID INT NOT NULL, -- ID của nội dung bị báo cáo
    Reason NVARCHAR(500) NOT NULL,
    Status NVARCHAR(20) DEFAULT 'Pending', -- Pending, Resolved, Rejected
    ReviewedBy INT NULL,
    ReviewNote NVARCHAR(500),
    ReportedAt DATETIME DEFAULT GETDATE(),
    ReviewedAt DATETIME NULL,
    FOREIGN KEY (ReporterID) REFERENCES Users(UserID),
    FOREIGN KEY (ReviewedBy) REFERENCES Users(UserID)
);

-- =============================================
-- 7. BẢNG THANH TOÁN & GÓI VIP
-- =============================================

-- Bảng gói VIP
CREATE TABLE VIPPackages (
    PackageID INT PRIMARY KEY IDENTITY(1,1),
    PackageName NVARCHAR(100) NOT NULL,
    Duration INT NOT NULL, -- Số ngày
    Price DECIMAL(10,2) NOT NULL,
    Description NVARCHAR(500),
    IsActive BIT DEFAULT 1,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- Bảng giao dịch thanh toán
CREATE TABLE Transactions (
    TransactionID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    PackageID INT NOT NULL,
    Amount DECIMAL(10,2) NOT NULL,
    PaymentMethod NVARCHAR(50), -- MoMo, ZaloPay, Banking, etc.
    TransactionCode NVARCHAR(100) UNIQUE,
    Status NVARCHAR(20) DEFAULT 'Pending', -- Pending, Success, Failed, Cancelled
    PaidAt DATETIME NULL,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (PackageID) REFERENCES VIPPackages(PackageID)
);

-- =============================================
-- 8. BẢNG THÔNG BÁO
-- =============================================

CREATE TABLE Notifications (
    NotificationID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    Type NVARCHAR(50) NOT NULL, -- NewContent, Approval, RoleUpgrade, CommentReply, etc.
    Title NVARCHAR(255) NOT NULL,
    Content NVARCHAR(500),
    RelatedURL NVARCHAR(500), -- Link đến nội dung liên quan
    IsRead BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE
);

-- =============================================
-- 9. BẢNG BANNER QUẢNG CÁO
-- =============================================

CREATE TABLE Banners (
    BannerID INT PRIMARY KEY IDENTITY(1,1),
    Title NVARCHAR(255),
    ImageURL NVARCHAR(500) NOT NULL,
    LinkURL NVARCHAR(500),
    Position NVARCHAR(50), -- Top, Sidebar, Bottom, etc.
    StartDate DATETIME,
    EndDate DATETIME,
    IsActive BIT DEFAULT 1,
    DisplayOrder INT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- =============================================
-- 10. BẢNG ĐIỂM THƯỞNG
-- =============================================

CREATE TABLE PointTransactions (
    PointTransactionID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    Points INT NOT NULL, -- Số điểm (+/-)
    Type NVARCHAR(50), -- WatchMovie, ReadSeries, Comment, Redeem, etc.
    Description NVARCHAR(255),
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE
);

-- Bảng quà đổi điểm
CREATE TABLE Rewards (
    RewardID INT PRIMARY KEY IDENTITY(1,1),
    RewardName NVARCHAR(255) NOT NULL,
    Description NVARCHAR(500),
    PointCost INT NOT NULL,
    StockQuantity INT DEFAULT 0,
    IsActive BIT DEFAULT 1,
    CreatedAt DATETIME DEFAULT GETDATE()
);

-- Bảng lịch sử đổi quà
CREATE TABLE RewardRedemptions (
    RedemptionID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    RewardID INT NOT NULL,
    PointsSpent INT NOT NULL,
    Status NVARCHAR(20) DEFAULT 'Pending', -- Pending, Completed, Cancelled
    RedeemedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (RewardID) REFERENCES Rewards(RewardID)
);

    StatID INT PRIMARY KEY IDENTITY(1,1),
    StatDate DATE NOT NULL UNIQUE,

    TotalUsers INT DEFAULT 0,
    TotalMovies INT DEFAULT 0,
    TotalSeries INT DEFAULT 0,

);

    ErrorMessage NVARCHAR(MAX),
    BackupDate DATETIME DEFAULT GETDATE()
);

-- =============================================
-- 13. BẢNG THEO DÕI TÁC PHẨM (FOLLOW/SUBSCRIBE)
-- =============================================

CREATE TABLE Follows (
    FollowID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    ContentType NVARCHAR(20) NOT NULL, -- Movie, Series
    ContentID INT NOT NULL, -- MovieID hoặc SeriesID
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    UNIQUE(UserID, ContentType, ContentID)
);

-- =============================================
-- 14. BẢNG LOG HOẠT ĐỘNG HỆ THỐNG (AUDIT LOG)
-- =============================================

CREATE TABLE ActivityLogs (
    LogID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NULL, -- NULL nếu là hệ thống tự động
    Action NVARCHAR(100) NOT NULL, -- Login, Logout, Upload, Delete, Update, Comment, etc.
    TableName NVARCHAR(50), -- Bảng bị tác động
    RecordID INT, -- ID của bản ghi bị tác động
    OldValue NVARCHAR(MAX), -- Giá trị cũ (JSON)
    NewValue NVARCHAR(MAX), -- Giá trị mới (JSON)
    IPAddress NVARCHAR(50),
    UserAgent NVARCHAR(500),
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);

-- =============================================
-- 15. BẢNG TIN NHẮN/CHAT HỖ TRỢ
-- =============================================

CREATE TABLE SupportTickets (
    TicketID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    Subject NVARCHAR(255) NOT NULL,
    Status NVARCHAR(20) DEFAULT 'Open', -- Open, InProgress, Resolved, Closed
    Priority NVARCHAR(20) DEFAULT 'Normal', -- Low, Normal, High, Urgent
    AssignedTo INT NULL, -- Admin xử lý
    CreatedAt DATETIME DEFAULT GETDATE(),
    UpdatedAt DATETIME DEFAULT GETDATE(),
    ClosedAt DATETIME NULL,
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    FOREIGN KEY (AssignedTo) REFERENCES Users(UserID)
);

CREATE TABLE SupportMessages (
    MessageID INT PRIMARY KEY IDENTITY(1,1),
    TicketID INT NOT NULL,
    SenderID INT NOT NULL,
    Message NVARCHAR(MAX) NOT NULL,
    AttachmentURL NVARCHAR(500),
    IsAdminReply BIT DEFAULT 0,
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (TicketID) REFERENCES SupportTickets(TicketID) ON DELETE CASCADE,
    FOREIGN KEY (SenderID) REFERENCES Users(UserID)
);

-- 17. BẢNG LƯỢT LIKE/DISLIKE BÌNH LUẬN
-- =============================================

    ReactionID INT PRIMARY KEY IDENTITY(1,1),

    UserID INT NOT NULL,
    ReactionType NVARCHAR(10) NOT NULL, -- Like, Dislike

);

CREATE TABLE SeriesCommentReactions (
    ReactionID INT PRIMARY KEY IDENTITY(1,1),

    UserID INT NOT NULL,
    ReactionType NVARCHAR(10) NOT NULL, -- Like, Dislike
    CreatedAt DATETIME DEFAULT GETDATE(),
    FOREIGN KEY (CommentID) REFERENCES SeriesComments(CommentID) ON DELETE CASCADE,
    FOREIGN KEY (UserID) REFERENCES Users(UserID) ON DELETE CASCADE,
    UNIQUE(CommentID, UserID)
);

-- =============================================
-- 18. BẢNG PHIM/TRUYỆN LIÊN QUAN (RELATED CONTENT)
-- =============================================

CREATE TABLE RelatedMovies (
    RelatedID INT PRIMARY KEY IDENTITY(1,1),
    MovieID INT NOT NULL,
    RelatedMovieID INT NOT NULL,
    RelationType NVARCHAR(20) DEFAULT 'Similar', -- Similar, SameDirector, SameActor, Sequel
    DisplayOrder INT DEFAULT 0,
    FOREIGN KEY (MovieID) REFERENCES Movies(MovieID) ON DELETE CASCADE,

);

CREATE TABLE RelatedSeries (
    RelatedID INT PRIMARY KEY IDENTITY(1,1),
    SeriesID INT NOT NULL,
    RelatedSeriesID INT NOT NULL,
    RelationType NVARCHAR(20) DEFAULT 'Similar', -- Similar, SameAuthor, SameGenre, SpinOff
    DisplayOrder INT DEFAULT 0,
    FOREIGN KEY (SeriesID) REFERENCES Series(SeriesID) ON DELETE CASCADE,
    FOREIGN KEY (RelatedSeriesID) REFERENCES Series(SeriesID),
    UNIQUE(SeriesID, RelatedSeriesID)
);

-- =============================================
-- 24. BẢNG PHIẾU GIẢM GIÁ/COUPON
-- =============================================

-- BƯỚC 1: Kiểm tra các phim có thể trùng lặp
PRINT N'=== BƯỚC 1: Kiểm tra phim trùng lặp ===';
SELECT 
    m1.MovieID as MovieID1,
    m1.Title as Title1,
    m1.Slug as Slug1,
    m1.CreatedAt as CreatedAt1,
    (SELECT COUNT(*) FROM dbo.MovieEpisodes WHERE MovieID = m1.MovieID) as Episodes1,
    m2.MovieID as MovieID2,
    m2.Title as Title2,
    m2.Slug as Slug2,
    m2.CreatedAt as CreatedAt2,
    (SELECT COUNT(*) FROM dbo.MovieEpisodes WHERE MovieID = m2.MovieID) as Episodes2
FROM dbo.Movies m1
CROSS JOIN dbo.Movies m2
WHERE m1.MovieID < m2.MovieID
  AND (
    m1.Title = m2.Title
    OR m1.Slug LIKE m2.Slug + '%'
    OR m2.Slug LIKE m1.Slug + '%'
  )
ORDER BY m1.CreatedAt DESC;
GO

-- BƯỚC 2: Xem chi tiết tất cả phim và tập phim
PRINT N'=== BƯỚC 2: Chi tiết tất cả phim ===';
SELECT 
    m.MovieID,
    m.Title,
    m.Slug,
    m.Status,
    m.CreatedAt,
    COUNT(me.EpisodeID) as EpisodeCount,
    STRING_AGG('Tập ' + CAST(me.EpisodeNumber AS VARCHAR) + ': ' + ISNULL(me.Title, 'N/A'), ' | ') WITHIN GROUP (ORDER BY me.EpisodeNumber) as EpisodeList
FROM dbo.Movies m
LEFT JOIN dbo.MovieEpisodes me ON m.MovieID = me.MovieID
GROUP BY m.MovieID, m.Title, m.Slug, m.Status, m.CreatedAt
ORDER BY m.CreatedAt DESC;
GO

    SELECT * FROM INFORMATION_SCHEMA.COLUMNS 
    WHERE TABLE_NAME = 'Chapters' AND COLUMN_NAME = 'ImageCount'
)
BEGIN
    -- Cập nhật ImageCount nếu bảng ChapterImages đã tồn tại
    IF EXISTS (SELECT * FROM sys.tables WHERE name = 'ChapterImages')
    BEGIN
        UPDATE c
        SET c.ImageCount = ISNULL(img_cnt.cnt, 0)
        FROM Chapters c
        LEFT JOIN (
            SELECT ChapterID, COUNT(*) as cnt
            FROM ChapterImages
            GROUP BY ChapterID
        ) img_cnt ON c.ChapterID = img_cnt.ChapterID;
        
        PRINT N'   ✅ Đã cập nhật ImageCount từ bảng ChapterImages';
    END
    ELSE
    BEGIN
        -- Đảm bảo tất cả chapters có ImageCount = 0
        UPDATE Chapters SET ImageCount = 0 WHERE ImageCount IS NULL;
        PRINT N'   ✅ Đã set ImageCount = 0 cho tất cả chapters';
    END
END

-- 4. CẬP NHẬT UNIQUE CONSTRAINT CHO MovieEpisodes
-- =============================================
PRINT N'';
PRINT N'[4/4] Kiểm tra unique constraint cho MovieEpisodes...';

-- Kiểm tra xem có unique constraint (MovieID, EpisodeNumber) không
-- Nếu có SeasonNumber, cần unique (MovieID, SeasonNumber, EpisodeNumber)
IF EXISTS (
    SELECT * FROM INFORMATION_SCHEMA.COLUMNS 
    WHERE TABLE_NAME = 'MovieEpisodes' AND COLUMN_NAME = 'SeasonNumber'
)
BEGIN
    -- Kiểm tra xem đã có unique constraint với SeasonNumber chưa
    IF NOT EXISTS (
        SELECT * FROM sys.indexes 
        WHERE object_id = OBJECT_ID('MovieEpisodes')
        AND name = 'UQ_MovieEpisodes_MovieID_SeasonNumber_EpisodeNumber'
    )
    BEGIN
        -- Xóa constraint cũ nếu có (MovieID, EpisodeNumber)
        IF EXISTS (
            SELECT * FROM sys.indexes 
            WHERE object_id = OBJECT_ID('MovieEpisodes')
            AND name LIKE 'UQ_MovieEpisodes%'
        )
        BEGIN
            DECLARE @constraintName NVARCHAR(255);
            SELECT @constraintName = name 
            FROM sys.indexes 
            WHERE object_id = OBJECT_ID('MovieEpisodes')
            AND name LIKE 'UQ_MovieEpisodes%';
            
            EXEC('ALTER TABLE MovieEpisodes DROP CONSTRAINT ' + @constraintName);
            PRINT N'   → Đã xóa constraint cũ: ' + @constraintName;
        END
        
        -- Tạo unique constraint mới với SeasonNumber
        ALTER TABLE MovieEpisodes
        ADD CONSTRAINT UQ_MovieEpisodes_MovieID_SeasonNumber_EpisodeNumber 
        UNIQUE (MovieID, SeasonNumber, EpisodeNumber);
        
        PRINT N'   ✅ Đã tạo unique constraint với SeasonNumber';
    END
    ELSE
    BEGIN
        PRINT N'   ⚠️ Unique constraint với SeasonNumber đã tồn tại';
    END
END
GO
